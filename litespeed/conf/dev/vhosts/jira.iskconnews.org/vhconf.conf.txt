docroot $SERVER_ROOT/origin.iskconnews.org/www
vhdomain origin.iskconnews.org
enablegzip 1
enablebr 1
enableipgeo 1

accesslog access.log {
    useserver 0
    keepdays 7
    compressarchive 1
}

index  {
    useserver 0
    indexfiles index.php, index.html
    autoindex 0
}

context / {
    type redirect
    externalredirect 1
    statuscode 301
    location https://iskconnews.atlassian.net/jira/
}

errorlog $VH_ROOT/logs/error.log {
    keepdays 7
    useserver 0
    loglevel ERROR
    compressarchive 1
}

scripthandler  {
    add lsapi:lsphp php
}

rewrite  {
    enable 1
    autoloadhtaccess 1
    unknownkeywords rewriteengine On
    unknownkeywords rewritebase /
    unknownkeywords rewriterule ^/index\.php$ - [L]
    unknownkeywords rewritecond %{REQUEST_FILENAME} !-f
    unknownkeywords rewritecond %{REQUEST_FILENAME} !-d
    unknownkeywords rewriterule . /index.php [L]
    rules <<<MY_END
rewriteengine On
rewritebase /
rewriterule ^/index\.php$ - [L]
rewritecond %{REQUEST_FILENAME} !-f
rewritecond %{REQUEST_FILENAME} !-d
rewriterule . /index.php [L]
    MY_END
}
