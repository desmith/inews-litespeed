enableipgeo 1
enablegzip 1
docroot $SERVER_ROOT/prod.iskconnews.org/www
vhdomain iskconnews.org
vhaliases prod.iskconnews.org, www.iskconnews.org, jira.iskconnews.org
enablebr 1

accesslog access.log {
    useserver 0
    keepdays 7
    compressarchive 1
}

errorlog $VH_ROOT/logs/error.log {
    loglevel ERROR
    keepdays 7
    compressarchive 1
    useserver 0
}

scripthandler  {
    add lsapi:lsphp php
}

rewrite  {
    enable 1
    autoloadhtaccess 1
    unknownkeywords rewriteengine On
    unknownkeywords rewritebase /
    unknownkeywords rewriterule ^/index\.php$ - [L]
    unknownkeywords rewritecond %{REQUEST_FILENAME} !-f
    unknownkeywords rewritecond %{REQUEST_FILENAME} !-d
    unknownkeywords rewriterule . /index.php [L]
    rules <<<MY_END
rewriteengine On
rewritebase /
rewriterule ^/index\.php$ - [L]
rewritecond %{REQUEST_FILENAME} !-f
rewritecond %{REQUEST_FILENAME} !-d
rewriterule . /index.php [L]
    MY_END
}

index  {
    useserver 0
    indexfiles index.php, index.html
    autoindex 0
}
