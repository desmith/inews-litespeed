docroot $SERVER_ROOT/dev.iskconnews.org/www
vhdomain dev.iskconnews.org
vhaliases
enablegzip 1
enablebr 1
enableipgeo 1

errorlog $VH_ROOT/logs/error.log {
    useserver 0
    keepdays 7
    compressarchive 1
    loglevel ERROR
}

index  {
    useserver 0
    indexfiles index.php, index.html
    autoindex 0
}

scripthandler  {
    add lsapi:lsphp php
}

rewrite  {
    enable 1
    autoloadhtaccess 1
    rules <<<MY_END
rewriteengine On
rewritebase /
rewriterule ^/index\.php$ - [L]
rewritecond %{REQUEST_FILENAME} !-f
rewritecond %{REQUEST_FILENAME} !-d
rewriterule . /index.php [L]
MY_END
}

accesslog access.log {
    compressarchive 1
    useserver 0
    keepdays 7
}
