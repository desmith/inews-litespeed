setuidmode 2
restrained 1
vhroot $SERVER_ROOT/$VH_NAME/
configfile $SERVER_ROOT/conf/vhosts/$VH_NAME/vhconf.conf
allowsymbollink 1
chrootmode 0
enablescript 1

virtualhostconfig  {
    docroot $VH_ROOT/public/
    enablegzip 1

    hotlinkctrl  {
        allowdirectaccess 1
        onlyself 1
        enablehotlinkctrl 0
        suffixes gif, jpeg, jpg
    }

    accesscontrol  {
        allow *
    }

    index  {
        indexfiles index.html
        autoindex 0
        autoindexuri /_autoindex/default.php
        useserver 0
    }

    awstats  {
        workingdir $VH_ROOT/awstats
        awstatsuri /awstats/
        sitedomain localhost
        sitealiases 127.0.0.1 localhost
        updatemode 0
        updateinterval 86400
        updateoffset 0
        securedconn 0
    }

    expires  {
        enableexpires 1
    }

    accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
        logheaders 3
        rollingsize 500M
        compressarchive 0
        useserver 0
        keepdays 30
    }

    errorlog  {
        useserver 1
    }

    context / {
        maxconns 5
        location $VH_ROOT/
        type rails
        accesscontrol 
        adddefaultcharset off
        railsenv 1

        rewrite  {
            unknownkeywords rewritefile .htaccess
            rules rewritefile .htaccess
        }
    }

    rewrite  {
        enable 0
        loglevel 0
    }
}
